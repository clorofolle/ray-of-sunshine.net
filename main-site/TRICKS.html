Overlay when hovering on an image

More Info

CSS

 

.image-container {
    position: relative;
    width: 200px;
    margin: auto;
}
                    
.image-container .overlay {
    position: absolute;
    bottom: 0;
    left: 0;
    right: 0;
    background-color: rgba(111, 229, 117, 0.5);
    overflow: hidden;
    width: 100%;
    height: 20px;
    transition: .5s ease;
}
                    
.image-container:hover .overlay {
    height: 50%;
}

HTML

 

<div class="image-container" style="margin-bottom: 3%;">
    <img src="/images/my-art/pixel/Raygif.gif" alt="Description">
    <div class="overlay">More Info</div>
</div>

Animated loading spinner

CSS

 

.spinner {
    border: 4px solid rgba(0,0,0,.1);
    width: 36px;
    height: 36px;
    border-radius: 50%;
    border-left-color: rgb(8, 169, 11);
    animation: spin 1s ease infinite;
}

@keyframes spin {
    0% { transform: rotate(0deg); }
    100% { transform: rotate(360deg); }
}

HTML

 

<div class="spinner"></div>

Image gets squished on click!

CSS

 

.pet {
    display: inline-block;
    transition: transform 0.3s ease;
    cursor: pointer;
}
                    
.pet.squash-stretch {
    transform: scaleX(1.2) scaleY(0.8); /* Squash and stretch */
}

HTML

 

<div class="pet" id="pet">
    <img src="/images/my-art/pixel/rayplush.png">
</div>

JS

 

<script>
const pet = document.getElementById('pet');
                    
pet.addEventListener('click', function() {
    pet.classList.add('squash-stretch');
                    
// Remove the class after the animation is done
setTimeout(() => pet.classList.remove('squash-stretch'), 300); // 300ms = duration of the animation
                            });
</script>
